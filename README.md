<script>
  document.addEventListener('DOMContentLoaded', function() {
    // Animate visitor counter
    const counter = document.getElementById('visitor-counter');
    let count = 0;
    const target = Math.floor(Math.random() * 1000) + 1500;
    const increment = target / 100;
    
    function updateCounter() {
      if (count < target) {
        count += increment;
        counter.textContent = Math.floor(count);
        setTimeout(updateCounter, 20);
      } else {
        counter.textContent = target;
      }
    }
    
    updateCounter();
    
    // Create matrix effect
    const matrixBg = document.getElementById('matrix-bg');
    const chars = '01';
    
    function createMatrixStream() {
      const stream = document.createElement('div');
      stream.style.cssText = `
        position: absolute;
        top: -100px;
        left: ${Math.random() * 100}vw;
        color: #0f0;
        font-size: 20px;
        font-family: monospace;
        opacity: ${Math.random() * 0.3 + 0.1};
        animation: fall ${Math.random() * 5 + 3}s linear infinite;
        pointer-events: none;
        z-index: 1;
      `;
      
      let text = '';
      for(let i = 0; i < 15; i++) {
        text += chars.charAt(Math.floor(Math.random() * chars.length)) + '<br>';
      }
      
      stream.innerHTML = text;
      matrixBg.appendChild(stream);
      
      setTimeout(() => {
        stream.remove();
      }, 5000);
    }
    
    // Add fall animation to style
    const style = document.createElement('style');
    style.textContent = `
      @keyframes fall {
        from { transform: translateY(-100px); }
        to { transform: translateY(100vh); }
      }
    `;
    document.head.appendChild(style);
    
    // Start matrix streams
    setInterval(createMatrixStream, 150);
    
    // Add hover effect to project cards
    const projectCards = document.querySelectorAll('[style*="project-glow"]');
    projectCards.forEach(card => {

      });

        this.style.transform = 'translateY(0) scale(1)';
      });
    });
  });
